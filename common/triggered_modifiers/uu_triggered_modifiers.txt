# Province Triggered modifiers are here.
# These are added to provinces through the add_province_triggered_modifier effect

uu_FC_emperor_freeest = {
	potential = {
		has_reform = free_city
	}

	trigger = {
		has_reform = free_city
		has_country_flag = uu_freeest_city_trigger
	}
	diplomatic_reputation = 1
	diplomatic_annexation_cost = -0.15
}

uu_tim_controls_herat_tmod = {
	potential = {
		has_country_modifier = uu_tim_declared_for_herat_mod
		dynasty = "Timurid"
		NOT = { tag = TIM }
	}

	trigger = {
		controls = 446
		446 = { has_siege = no }
		has_country_modifier = uu_tim_declared_for_herat_mod
	}
	land_morale = 0.1
	stability_cost_modifier = -0.25
	global_tax_income = 48
}

uu_high_absolutism_tmod = {
	potential = {
		OR = {
			current_age = age_of_absolutism
			current_age = age_of_revolutions
		}
	}

	trigger = {
		absolutism = 99
	}
	all_power_cost = -0.05
	can_revoke_parliament_seats = yes
	allow_free_estate_privilege_revocation = yes
}

uu_ai_native_nomigrate_tmod = {
	potential = {
		ai = yes
		NOT = { has_reform = native_settle_down_reform }
		government = native
		capital_scope = {
			OR = {
				region = brazil_region
				region = upper_peru_region
				region = cascadia_region
				region = california_region
			}
		}
	}

	trigger = {
		NOT = { has_reform = native_settle_down_reform }
	}
	migration_cost = 5.0 #this should deter them from migrating so much in the areas with gold
}

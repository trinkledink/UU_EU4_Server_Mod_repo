country_event = {
	id = uu_kitara.3

	title = "uu_kitara.3.t"
	desc = "uu_kitara.3.d"
	picture = GREAT_BUILDING_eventPicture
	goto = uu_kitara_urbanization_target

	trigger = {
		any_owned_province = {
			NOT = { has_province_flag = uu_kitara_urbanization_target }
		}
	}
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			4064 = {							#Great Lakes
				random_trade_node_member_province = {
					limit = {
						is_overseas = no
						is_city = yes
						NOT = { has_province_flag = uu_kitara_urbanization_target }
						owned_by = ROOT
					}
					save_event_target_as = uu_kitara_urbanization_target
					set_province_flag = uu_kitara_urbanization_target
				}
			}
		}
	}
	option = {
		name = "uu_kitara.3.a"
		event_target:uu_kitara_urbanization_target = {
			add_base_tax = 2
			add_base_production = 2
			add_base_manpower = 2
		}
	}
}

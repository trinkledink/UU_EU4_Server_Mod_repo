#A Wattasid Conspiracy?			UU Changed
country_event = {
	id = flavor_mor.6
	title = flavor_mor.6.t
	desc = flavor_mor.6.duu
	picture = ACCUSATION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = MOR
		has_country_flag = mor_death_of_yahya
		NOT = { is_year = 1490 }
	}
	
	mean_time_to_happen = {
		months = 126
	}
	
	option = {
		name = flavor_mor.6.a #Kill them all!
		ai_chance = { factor = 1 }
		if = {
			limit = { 				#Wat ruler, Wat heir
				dynasty = "Wattasid" 
				heir_has_ruler_dynasty = yes
			}
			kill_ruler = yes
			define_ruler = { dynasty = "Marinid" }
			kill_heir = { allow_new_heir = no }
		} 	
		else_if = {					#Wat ruler, Mar heir
			limit = { 
				dynasty = "Wattasid" 
				OR = {
					heir_has_ruler_dynasty = no
					has_heir = no
				}
			}
			kill_ruler = yes
		} 
		else = { 					#Mar ruler, any heir
			add_stability = -1 
		}	
		custom_tooltip = uumor_add_stats_tt
		hidden_effect = {
			change_adm = 2
			change_mil = 2
		}
		add_country_modifier = {
			name = uumor_unrest_mod
			duration = 1825
		}
		country_event = { id = flavor_mor.7 days = 30 }
	}
	
	option = {
		name = flavor_mor.6.b #They remain our trusted advisors.
		ai_chance = { factor = 0 }
		add_stability = 1
		set_country_flag = mor_wattasid_power
	}
	
}

#The Sudanese Expedition
country_event = {
	id = flavor_mor.9
	title = flavor_mor.9.t
	desc = flavor_mor.9.d
	picture = EXPLORERS_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			AND = {
				NOT = { has_dlc = "Golden Century" }
				tag = MOR
				is_year = 1550
				OR = {
					owns_core_province = 2466
					owns_core_province = 2460
				}
				1127 = {
					is_empty = yes
				}
				NOT = { owns = 1128 }
			}
			has_country_flag = mor_sudanese_expedition
		}
	}
	
	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					owns = 1128
					owns = 1129
					owns = 1132
					owns = 2247
				}
				set_country_flag = mor_crosses_sahara
			}
		}
	}
	
	mean_time_to_happen = {
		months = 150
	}
	
	option = {
		name = flavor_mor.9.a #Let us cross the Sahara!
		ai_chance = { factor = 1 }
		1127 = {
			add_core = MOR
			cede_province = MOR
		}
		every_province = {
			limit = {
				OR = {
					province_id = 1128
					province_id = 1129
					province_id = 1132
					province_id = 2247
				}
			}
			add_permanent_claim = MOR
		}
		every_country = {
			limit = { has_country_flag = mor_crosses_sahara }
			country_event = { id = flavor_mor.10 }
		}
		
	}	
}
